cmake_minimum_required(VERSION 3.15)
project(asteroids_enemies_sdl2 CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


find_package(SDL2 REQUIRED)

add_executable(asteroids
    src/main.cpp
    src/Enemy.cpp
    src/Enemy.h
)


target_link_libraries(asteroids PRIVATE SDL2::SDL2 SDL2::SDL2main)


if (MSVC)
    target_link_options(asteroids PRIVATE /SUBSYSTEM:CONSOLE)
endif()
